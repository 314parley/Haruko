 Example plugin:
 
<?php
class plugin_TestMenu implements IPlugin {
	public function load($conn)
	{
	
	}
	
	public function runHook($hook, $params)
	{
		if ($hook == "menu")
		{
			return '<li><a href="?/test" target="main">Test menu</a></li>';
		}
		if ($hook == "panel")
		{
			if ($params == "/test")
			{
				return "<b>Test</b>";
			}
		}
	}
}
?>
